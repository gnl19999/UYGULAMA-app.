import SwiftUI // MARK: - Hex Renk Desteği için Eklenti (Hatanın düzeltildiği yer burasıdır) // Bu uzantı, Color nesnelerinin kullanıldığı her yerden önce tanımlanmalıdır. extension Color { init(hex: String) { let hex = hex.trimmingCharacters(in: CharacterSet.alphanumerics.inverted) var int: UInt64 = 0 Scanner(string: hex).scanHexInt64(&int) let a, r, g, b: UInt64 switch hex.count { case 3: // RGB (12-bit) (a, r, g, b) = (255, (int >> 8) * 17, (int >> 4 & 0xF) * 17, (int & 0xF) * 17) case 6: // RGB (24-bit) (a, r, g, b) = (255, int >> 16, int >> 8 & 0xFF, int & 0xFF) case 8: // ARGB (32-bit) (a, r, g, b) = (int >> 24, int >> 16 & 0xFF, int >> 8 & 0xFF, int & 0xFF) default: (a, r, g, b) = (1, 1, 1, 0) } self.init( .sRGB, red: Double(r) / 255, green: Double(g) / 255, blue: Double(b) / 255, opacity: Double(a) / 255 ) } } // MARK: - Veri Modeli struct SubscriptionPlan: Identifiable { let id = UUID() let title: String let price: String let perMonth: String let isBestValue: Bool } // MARK: SubscriptionPlan(title: "6 aylık abonelik", price: "₺549,99", perMonth: "₺91.67 / ay", isBestValue: false), SubscriptionPlan(title: "Yıllık abonelik", price: "₺749,99", perMonth: "₺62.50 / ay", isBestValue: true) ] // Seçili planı takip eden değişken @State private var selectedPlanId: UUID? // Custom renkler private let lightBlue = Color(hex: "4FC3F7") private let brightYellow = Color(hex: "FDD835") init() { // En Avantajlı planı varsayılan olarak seç _selectedPlanId = State(initialValue: plans.first(where: { $0.isBestValue })?.id ?? plans.last?.id) } var body: some View { ZStack { // Arka Plan (Mavi Gradient) LinearGradient(gradient: Gradient(colors: [lightBlue, Color.white]), startPoint: .top, endPoint: .bottom) .edgesIgnoringSafeArea(.all) VStack(spacing: 20) { // --- Üst Kısım --- Spacer().frame(height: 20) Image(systemName: "crown.fill") .resizable() .scaledToFit() .frame(height: 50) .foregroundColor(brightYellow) .shadow(radius: 2) Text("İngilizce kelime öğrenmenin\nen etkili yolu") .font(.title2) .fontWeight(.bold) .multilineTextAlignment(.center) .foregroundColor(.white) // --- Özellik Listesi (Beyaz Kutu) --- VStack(alignment: .leading, spacing: 15) { FeatureRow(icon: "crown.fill", text: "Premium'a özel paketleri çalışmaya başlayın", iconColor: brightYellow) FeatureRow(icon: "plus.app.fill", text: "Kendi kelime paketlerinizi oluşturun", iconColor: lightBlue) FeatureRow(icon: "equal.square.fill", text: "Eş anlamlı çalışması ile daha fazla kelime öğrenin", iconColor: lightBlue) FeatureRow(icon: "lightbulb.fill", text: "2500'den fazla kelimeyi başlatıldı: \(String(describing: selectedPlanId))") }) { Text("Premium'a Yükselt") .font(.title3) .fontWeight(.heavy) .foregroundColor(.black) .frame(maxWidth: .infinity) .padding() .background(brightYellow) .cornerRadius(30) } .padding(.horizontal) .padding(.bottom, 20) } } } } // MARK: - Yardımcı Görünüm: Özellik Satırı struct FeatureRow: View { let icon: String let text: .fixedSize(horizontal: false, vertical: true) } } } // MARK: - Yardımcı Görünüm: Plan Kartı struct PlanCard: View { let plan: SubscriptionPlan let isSelected: Bool let accentColor: Color let
